**1. getDaemonClipboardEntries(limit?: number, offset?: number)**

```typescript
export async function getDaemonClipboardEntries(
  limit = 50,
  offset = 0
): Promise<ClipboardItemsResult> {
  const response = await daemonClient.request<ClipboardEntriesResponse>('/clipboard/entries', {
    searchParams: { limit, offset },
  })
  if (response.status === 'not_ready') return { status: 'not_ready' }
  return { status: 'ready', items: response.entries.map(transformProjectionToResponse) }
}
```

**2. getDaemonClipboardEntry(entryId: string)**

```typescript
export async function getDaemonClipboardEntry(
  entryId: string
): Promise<ClipboardItemResponse | null> {
  const response = await daemonClient.request<ClipboardEntriesResponse>(
    `/clipboard/entries/${entryId}`
  )
  if (response.status === 'not_ready' || response.entries.length === 0) return null
  return transformProjectionToResponse(response.entries[0])
}
```

**3. deleteDaemonClipboardEntry(entryId: string)**

```typescript
export async function deleteDaemonClipboardEntry(entryId: string): Promise<boolean> {

  return true
}
```
