```rust
// src-tauri/crates/uc-cli/src/daemon_client.rs

use std::fmt;
use std::path::{Path, PathBuf};
use std::time::Duration;

use anyhow::Result;
use reqwest::{Client, StatusCode};
use serde::{de::DeserializeOwned, Serialize};
use uc_daemon::api::auth::resolve_daemon_token_path;
use uc_daemon_client::http::exchange_cli_session_token;
use uc_daemon_client::DaemonConnectionState;
use uc_daemon_client::connection::DaemonConnectionInfo;
use tokio::sync::RwLock;

const DEFAULT_QUERY_TIMEOUT_SECS: u64 = 2;
const SETUP_ACTION_TIMEOUT_SECS: u64 = 15;

/// CLI daemon HTTP client using session exchange pattern.
/// Each command is a fresh process, so token exchange happens on every call.
/// Per D-07: No JWT file caching.
pub struct DaemonHttpClient {
    http: Client,
    /// Bearer token (from daemon.token file) — used only for /auth/connect exchange.
    bearer_token: String,
    /// The CLI's OS process ID, used for daemon PID whitelist registration.
    cli_pid: u32,
    /// In-memory session token for this command invocation.
    /// Exchanged fresh on first HTTP request via POST /auth/connect.
    session_token: RwLock<Option<String>>,
}

impl DaemonHttpClient {
    /// Create a new CLI daemon client.
    /// Reads bearer token from daemon.token file and captures current PID.
    pub fn new() -> Result<Self, DaemonClientError> {
        let bearer_token = resolve_bearer_token()?;
        let cli_pid = std::process::id();
        let http = Client::builder()
            .timeout(Duration::from_secs(DEFAULT_QUERY_TIMEOUT_SECS))
            .build()
            .map_err(|e| DaemonClientError::Initialization(anyhow::Error::new(e)))?;

        Ok(Self {
            http,
            bearer_token,
            cli_pid,
            session_token: RwLock::const_new(None),
        })
    }

    /// Exchange bearer token for session JWT via POST /auth/connect.
    /// Per D-07: Called fresh on every HTTP request (no file caching).
    async fn ensure_session_token(&self, connection_state: &DaemonConnectionState) -> Result<String, DaemonClientError> {
        // Check if we already have a valid session token
        {
            let guard = self.session_token.read().await;
            if let Some(token) = &*guard {
                return Ok(token.clone());
            }
        }

        // Exchange new token via /auth/connect
        let token = exchange_cli_session_token(
            &self.http,
            connection_state,
            self.cli_pid,
        )
        .await
        .map_err(|e| DaemonClientError::Initialization(anyhow::Error::new(e)))?;

        // Cache in memory for this command
        {
            let mut guard = self.session_token.write().await;
            *guard = Some(token.clone());
        }

        Ok(token)
    }

    /// Build an authorized request using the session JWT.
    async fn authorized_request(
        &self,
        connection_state: &DaemonConnectionState,
        method: reqwest::Method,
        path: &str,
    ) -> Result<reqwest::RequestBuilder, DaemonClientError> {
        let session_token = self.ensure_session_token(connection_state).await?;
        let connection = connection_state.get()
            .ok_or_else(|| DaemonClientError::Unreachable(anyhow::anyhow!("daemon connection info unavailable")))?;
        let url = format!("{}{}", connection.base_url, path);

        Ok(self.http
            .request(method, url)
            .header("Authorization", format!("Session {}", session_token)))
    }

    // All existing API methods (get_status, get_peers, etc.) rewritten to use authorized_request:

    pub async fn get_status(&self) -> Result<StatusResponse, DaemonClientError> {
        self.get_json("/status").await
    }

    pub async fn get_peers(&self) -> Result<Vec<PeerSnapshotDto>, DaemonClientError> {
        self.get_json("/peers").await
    }

    pub async fn get_paired_devices(&self) -> Result<Vec<PairedDeviceDto>, DaemonClientError> {
        self.get_json("/paired-devices").await
    }

    pub async fn get_setup_state(&self) -> Result<SetupStateResponse, DaemonClientError> {
        self.get_json("/setup/state").await
    }

    // Keep all other existing methods (start_setup_host, start_setup_join, etc.) unchanged
    // but update their implementation to use authorized_request pattern.

    // ... (rest of existing methods, adapted to use authorized_request) ...

    async fn get_json<T>(&self, connection_state: &DaemonConnectionState, path: &str) -> Result<T, DaemonClientError>
    where T: DeserializeOwned {
        let req = self.authorized_request(connection_state, reqwest::Method::GET, path).await?;
        let response = req.send().await.map_err(map_reqwest_error)?;
        decode_json_response(response).await
    }

    // ... other helper methods (post_json, post_without_body, etc.) ...
}
```
